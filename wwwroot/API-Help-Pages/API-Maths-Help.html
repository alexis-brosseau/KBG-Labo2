GET : Maths endpoint
List of possible query strings

? op = + & x = number & y = number
return {"op":"+","x":number,"y":number,"value":x+y}

? op = - & x = number & y = number
return {"op":"-","x":number,"y":number,"value":x-y}

? op = * & x = number & y = number
return {"op":"*","x":number,"y":number,"value":x*y}

? op = / & x = number & y = number
return {"op":"/","x":number,"y":number,"value":x/y}

? op = % & x = number & y = number
return {"op":"%","x":number,"y":number,"value":x%y}

? op = ! & n = integer
return {"op":"!","n":integer,"value":factorial(n)}

? op = p & n = integer
return {"op":"p","n":integer,"value":isPrime(n)}

? op = np & n = integer
return {"op":"np","n":integer,"value":nthPrime(n)}
